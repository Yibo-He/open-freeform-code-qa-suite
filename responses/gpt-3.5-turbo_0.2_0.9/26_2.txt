To fix the issue of the user not changing when navigating to another user, you need to use a watcher to update the `username` computed property whenever the `userToLoad` value changes. Here's how you can modify your code to achieve this:

```
<template>
  <div>{{username}}</div>
</template>

<script lang="ts">
import { computed, defineComponent, ref, getCurrentInstance, watch } from '@vue/composition-api';

export const useUsername = ({ user }) => {
  return { username: user.name };
};

export default defineComponent({
  setup(props, { root }) {
    const vm = getCurrentInstance();

    const userToLoad = computed(() => root.$route.params.userId);
    const listOfUsers = [
      { userId: 1, name: 'user1' },
      { userId: 2, name: 'user2' },
    ];

    const user = ref(listOfUsers.find((u) => u.userId === +userToLoad.value));
    if (!user.value) {
      return root.$router.push('/404');
    }

    watch(userToLoad, (newValue) => {
      const newUser = listOfUsers.find((u) => u.userId === +newValue);
      if (!newUser) {
        return root.$router.push('/404');
      }
      user.value = newUser;
    });

    const { username } = useUsername({ user });

    return { username };
  },
});

</script>
```
