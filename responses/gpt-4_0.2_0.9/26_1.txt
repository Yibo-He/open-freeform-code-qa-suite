The issue you're facing is due to the fact that Vue's `setup` function is only called once when the component is created. This means that when you navigate to a different user, the `setup` function is not called again, and therefore the user data is not updated.

To fix this, you can use a `watch` function to observe changes to the route parameters and update the user data accordingly. Here's how you can do it:

```javascript
<template>
  <div>{{username}}</div>
</template>

<script lang="ts">
import { computed, defineComponent, ref, getCurrentInstance, watch } from '@vue/composition-api';

export const useUsername = ({ user }) => {
  return { username: user.name };
};

export default defineComponent({
  setup(props, { root }) {
    const vm = getCurrentInstance();

    const userToLoad = computed(() => root.$route.params.userId);
    const listOfUsers = [
      { userId: 1, name: 'user1' },
      { userId: 2, name: 'user2' },
    ];

    const user = ref(listOfUsers.find((u) => u.userId === +userToLoad.value));

    watch(userToLoad, (newUserId) => {
      const newUser = listOfUsers.find((u) => u.userId === +newUserId);
      if (!newUser) {
        root.$router.push('/404');
      } else {
        user.value = newUser;
      }
    }, { immediate: true });

    const { username } = useUsername({ user: user.value });

    return { username };
  },
});
</
